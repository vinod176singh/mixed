def getSum(lst,tsum):
    lst.sort()
    left=0
    right=len(lst)-1
    while right>left:
        if lst[left]+lst[right]<tsum:
            left+=1
        elif lst[left]+lst[right]>tsum:
            right-=1
        else:
            if lst[left]+lst[right]==tsum
            return [left,right]
